# Case Study - The Dice Class

The Dice class created in C#

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Dice
{
    class Dice
    {

        protected const byte DICEMINIMUMFACEVALUE = 1;
        protected const byte DICEMAXIMUMFACEVALUE = 6;

        protected byte faceValue = DICEMINIMUMFACEVALUE;

        // random number generator with high quality seed value
        private Random random = new Random(int.Parse(Guid.NewGuid().ToString().Substring(0, 8), System.Globalization.NumberStyles.HexNumber));

        //public Dice() // not needed if the overloaded constructor has a default value
        //{

        //}

        public Dice(byte faceValue = DICEMAXIMUMFACEVALUE) // overloaded constructor
        {
            if(faceValue >= DICEMINIMUMFACEVALUE && faceValue <= DICEMAXIMUMFACEVALUE)
            {
                this.faceValue = faceValue;
            }
        }

        public byte Roll()
        {
            faceValue = Convert.ToByte(random.Next(DICEMINIMUMFACEVALUE, DICEMAXIMUMFACEVALUE + 1));
            return faceValue;
        }

        public void Draw()
        {
            Console.WriteLine(faceValue);
        }

        public byte GetFaceValue() { return faceValue; }
    }
}

The Dice class created in Kotlin

package dicepackage

import kotlin.random.Random

class Dice(faceValue: Short = 1){ // faceValue is just a passed argument not a property

    private val MAX_VALUE = 6
    private val MIN_VALUE = 1

    var faceValue: Short = 1
        private set(value) { // value is the value passed to the setter
        require(value in MIN_VALUE..MAX_VALUE) { "Invalid Dice Value" }
        field = value // field refers to faceValue
    }

    init{
        this.faceValue = faceValue // faceValue field must exist before init is called
    }

    fun draw(){
        println(faceValue)
    }

    fun roll(): Short {
        faceValue = Random.nextInt(MIN_VALUE, MAX_VALUE + 1).toShort()
        return faceValue
    }
}

# Case Study - The DiceDLX Class

The DiceDLX class created in C#

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace Dice
{
    class DiceDLX : Dice
    {
        protected uint[] rolledNumbers = new uint[DICEMAXIMUMFACEVALUE];

        protected void GenerateRollStatistics()
        {
            rolledNumbers[faceValue - 1]++;
        }

        public DiceDLX()
        {

        }

        public DiceDLX(byte faceValue) : base(faceValue)
        {

        }

        public new void Draw()
        {
            Console.WriteLine("*****");
            Console.WriteLine($"* {faceValue} *");
            Console.WriteLine("*****");
         }

        public new byte Roll()
        {
            base.Roll();
            GenerateRollStatistics();
            return faceValue;
        }

        public uint[] GetStatistics()
        {
            return rolledNumbers;
        }

    }
}
